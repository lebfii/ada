#include<stdio.h>

#define MAX 8

int board[MAX][MAX] = {0};

// Check if queen is safe
int safe(int row, int col) {

    int i, j;

    // Left side
    for(i = 0; i < col; i++) {

        if(board[row][i])
            return 0;
    }

    // Upper diagonal
    for(i = row, j = col; i >= 0 && j >= 0; i--, j--) {

        if(board[i][j])
            return 0;
    }

    // Lower diagonal
    for(i = row, j = col; i < MAX && j >= 0; i++, j--) {

        if(board[i][j])
            return 0;
    }

    return 1;
}

// Place queens
int queen(int col) {

    if(col >= MAX)
        return 1;

    for(int i = 0; i < MAX; i++) {

        if(safe(i, col)) {

            board[i][col] = 1;

            if(queen(col + 1))
                return 1;

            board[i][col] = 0; // backtrack
        }
    }

    return 0;
}

int main() {

    queen(0);

    for(int i = 0; i < MAX; i++) {

        for(int j = 0; j < MAX; j++) {

            printf("%d ", board[i][j]);
        }

        printf("\n");
    }

    return 0;
}
