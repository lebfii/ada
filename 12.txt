#include<stdio.h>

#define N 8

int board[N][N]={0};

int safe(int r,int c)
{
    int i,j;

    for(i=0;i<c;i++)
        if(board[r][i])
            return 0;

    for(i=r,j=c;i>=0&&j>=0;i--,j--)
        if(board[i][j])
            return 0;

    for(i=r,j=c;i<N&&j>=0;i++,j--)
        if(board[i][j])
            return 0;

    return 1;
}

int solve(int c)
{
    int i;

    if(c>=N)
        return 1;

    for(i=0;i<N;i++)
    {
        if(safe(i,c))
        {
            board[i][c]=1;

            if(solve(c+1))
                return 1;

            board[i][c]=0;
        }
    }

    return 0;
}

int main()
{
    int i,j;

    if(solve(0))
    {
        for(i=0;i<N;i++)
        {
            for(j=0;j<N;j++)
                printf("%d ",board[i][j]);

            printf("\n");
        }
    }
    else
        printf("No solution");

    return 0;
}
