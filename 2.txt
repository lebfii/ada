#include<stdio.h>

#define N 3
#define INF 999

int G[N][N] = {
    {0, 19, 8},
    {21, 0, 13},
    {15, 18, 0}
};

int main() {
    int selected[N] = {0};
    int edges = 0, min, x, y;
    int cost = 0;

    selected[0] = 1;

    int S[N][N] = {0};

    while(edges < N - 1) {
        min = INF;

        for(int i = 0; i < N; i++) {
            if(selected[i]) {
                for(int j = 0; j < N; j++) {
                    if(!selected[j] && G[i][j]) {
                        if(G[i][j] < min) {
                            min = G[i][j];
                            x = i;
                            y = j;
                        }
                    }
                }
            }
        }

        S[x][y] = G[x][y];
        S[y][x] = G[x][y];

        cost += G[x][y];
        selected[y] = 1;
        edges++;
    }

    printf("Spanning tree:\n");

    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            printf("%d\t", S[i][j]);
        }
        printf("\n");
    }

    printf("Minimum cost = %d", cost);

    return 0;
}
