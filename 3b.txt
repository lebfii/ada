#include<stdio.h>

#define MAX 10

int main() {

    int n, i, j, k;
    int a[MAX][MAX], p[MAX][MAX];

    printf("Enter number of nodes: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");

    for(i = 0; i < n; i++) {

        for(j = 0; j < n; j++) {

            scanf("%d", &a[i][j]);

            p[i][j] = a[i][j];
        }
    }

    // Path Matrix
    for(k = 0; k < n; k++) {

        for(i = 0; i < n; i++) {

            for(j = 0; j < n; j++) {

                if(p[i][k] && p[k][j]) {

                    p[i][j] = 1;
                }
            }
        }
    }

    printf("\nPath Matrix:\n");

    for(i = 0; i < n; i++) {

        for(j = 0; j < n; j++) {

            printf("%d ", p[i][j]);
        }

        printf("\n");
    }

    return 0;
}
