#include<stdio.h>

int main() {

    int n, i, j;
    float w[10], p[10], r[10];
    float cap, total = 0, temp;

    printf("Enter number of items: ");
    scanf("%d", &n);

    // Input
    for(i = 0; i < n; i++) {

        printf("Enter weight and profit: ");
        scanf("%f %f", &w[i], &p[i]);

        r[i] = p[i] / w[i];
    }

    printf("Enter capacity: ");
    scanf("%f", &cap);

    // Sort by ratio
    for(i = 0; i < n; i++) {

        for(j = i + 1; j < n; j++) {

            if(r[i] < r[j]) {

                temp = r[i];
                r[i] = r[j];
                r[j] = temp;

                temp = w[i];
                w[i] = w[j];
                w[j] = temp;

                temp = p[i];
                p[i] = p[j];
                p[j] = temp;
            }
        }
    }

    // Fill knapsack
    for(i = 0; i < n; i++) {

        if(w[i] <= cap) {

            total += p[i];
            cap -= w[i];
        }
        else {

            total += r[i] * cap;
            break;
        }
    }

    printf("Maximum profit = %.2f", total);

    return 0;
}
