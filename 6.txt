#include <stdio.h>

#define MAX 10

int a[MAX][MAX], n, indegre[MAX];

void find_indegre()
{
    int i, j, sum;

    for (j = 0; j < n; j++)
    {
        sum = 0;
        for (i = 0; i < n; i++)
            sum += a[i][j];

        indegre[j] = sum;
    }
}

void topology()
{
    int i, u, v;
    int t[MAX], s[MAX];
    int top = -1, k = 0;

    find_indegre();

    for (i = 0; i < n; i++)
    {
        if (indegre[i] == 0)
            s[++top] = i;
    }

    while (top != -1)
    {
        u = s[top--];
        t[k++] = u;

        for (v = 0; v < n; v++)
        {
            if (a[u][v] == 1)
            {
                indegre[v]--;
                if (indegre[v] == 0)
                    s[++top] = v;
            }
        }
    }

    printf("The topological sequence is:\n");

    for (i = 0; i < k; i++)
        printf("%d ", t[i]);

    printf("\n");
}

int main()
{
    int i, j;

    printf("Enter number of jobs: ");
    scanf("%d", &n);

    printf("\nEnter the adjacency matrix:\n");

    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
            scanf("%d", &a[i][j]);
    }

    topology();

    return 0;
}
